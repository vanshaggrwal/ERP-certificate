rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() {
      return request.auth != null;
    }

    function requesterUser() {
      return isSignedIn()
        ? get(/databases/$(database)/documents/users/$(request.auth.uid)).data
        : null;
    }

    function requesterRole() {
      return requesterUser() != null && requesterUser().role is string
        ? requesterUser().role.lower()
        : "";
    }

    function requesterCollegeCode() {
      return requesterUser() != null && requesterUser().collegeCode is string
        ? requesterUser().collegeCode.lower()
        : "";
    }

    function isSuperAdmin() {
      return requesterRole() == "superadmin";
    }

    function isCollegeAdmin() {
      return requesterRole() == "collegeadmin";
    }

    function isStudent() {
      return !isSuperAdmin() && !isCollegeAdmin() && isSignedIn();
    }

    function sameUser(userId) {
      return isSignedIn() && request.auth.uid == userId;
    }

    function sameAuthEmail(resourceEmail) {
      return isSignedIn() &&
        request.auth.token.email is string &&
        resourceEmail is string &&
        request.auth.token.email == resourceEmail;
    }

    function sameAuthEmailFromStudentDoc(studentData) {
      return sameAuthEmail(studentData.email) ||
        sameAuthEmail(studentData.OFFICIAL_DETAILS["EMAIL_ID"]) ||
        sameAuthEmail(studentData.OFFICIAL_DETAILS["EMAIL_ID."]);
    }

    function projectDocBelongsToRequester(projectDocId) {
      return ("" + projectDocId).lower().split("-")[0] == requesterCollegeCode();
    }

    function projectCodeBelongsToRequester(projectCode) {
      return ("" + projectCode).lower().split("/")[0] == requesterCollegeCode();
    }

    function studentBelongsToRequester(studentData, projectDocId) {
      return ("" + studentData.collegeCode).lower() == requesterCollegeCode() ||
        projectDocBelongsToRequester(projectDocId);
    }

    function ticketOwnedByRequester(ticketData) {
      return isSignedIn() && (
        (ticketData.createdByUid is string && ticketData.createdByUid == request.auth.uid) ||
        sameAuthEmail(ticketData.createdByEmail)
      );
    }

    match /users/{userId} {
      allow read: if sameUser(userId) || isSuperAdmin();
      allow create: if isSuperAdmin();
      allow update: if sameUser(userId) || isSuperAdmin();
      allow delete: if isSuperAdmin();
    }

    match /student_users/{userId} {
      allow read: if isSuperAdmin() ||
        sameUser(userId) ||
        (isSignedIn() &&
          (
            (resource.data.uid is string && resource.data.uid == request.auth.uid) ||
            sameAuthEmail(resource.data.email)
          ));
      allow create, update: if isSuperAdmin() || sameUser(userId);
      allow delete: if isSuperAdmin();
    }

    match /projectCodes/{projectCodeId} {
      allow read: if isSuperAdmin() ||
        (isCollegeAdmin() &&
          ("" + resource.data.collegeId).lower() == requesterCollegeCode());
      allow create, update, delete: if isSuperAdmin();
    }

    // certificateProjectEnrollments collection REMOVED — replaced by
    // students/{projectDocId}/certificate_enrollments/{enrollmentDocId}

    match /certificates/{certificateId} {
      allow read: if isSignedIn();
      allow create, update, delete: if isSuperAdmin();
    }

    match /organizations/{organizationId} {
      allow read: if isSignedIn();
      allow create, update, delete: if isSuperAdmin();
    }

    match /students/{projectDocId} {
      allow read: if isSuperAdmin() ||
        (isCollegeAdmin() && projectDocBelongsToRequester(projectDocId));
      allow create, update, delete: if isSuperAdmin();

      // Flat certificate_enrollments subcollection directly under project doc
      match /certificate_enrollments/{enrollmentDocId} {
        allow read: if isSuperAdmin() ||
          (isCollegeAdmin() &&
            projectDocBelongsToRequester(projectDocId)) ||
          (isSignedIn() &&
            sameAuthEmail(resource.data.email));
        allow create, update, delete: if isSuperAdmin();
      }

      match /students_list/{studentId} {
        allow read: if isSuperAdmin() ||
          (isCollegeAdmin() &&
            studentBelongsToRequester(resource.data, projectDocId)) ||
          sameAuthEmailFromStudentDoc(resource.data);
        allow create, update, delete: if isSuperAdmin();

        // Legacy nested path — kept for backward compat reads only
        match /certificate_enrollments/{certEnrollmentId} {
          allow read: if isSuperAdmin() ||
            (isCollegeAdmin() &&
              projectDocBelongsToRequester(projectDocId)) ||
            sameAuthEmailFromStudentDoc(
              get(/databases/$(database)/documents/students/$(projectDocId)/students_list/$(studentId)).data
            );
          allow create, update, delete: if isSuperAdmin();
        }
      }
    }

    // CollectionGroup rule for students_list — required for collectionGroup queries
    // (e.g. getAllStudents, getAllStudentsCount on Dashboard)
    match /{path=**}/students_list/{studentId} {
      allow read: if isSuperAdmin() ||
        (isCollegeAdmin() &&
          ("" + resource.data.collegeCode).lower() == requesterCollegeCode()) ||
        sameAuthEmailFromStudentDoc(resource.data);
      allow create, update, delete: if isSuperAdmin();
    }

    // CollectionGroup rule for certificate_enrollments — required for collectionGroup queries
    // (e.g. getCertificateEnrollmentCounts, softDeleteCertificate across all projects)
    match /{path=**}/certificate_enrollments/{enrollmentDocId} {
      allow read: if isSuperAdmin() ||
        (isCollegeAdmin() &&
          ("" + resource.data.collegeCode).lower() == requesterCollegeCode()) ||
        (isSignedIn() && sameAuthEmail(resource.data.email));
      allow create, update, delete: if isSuperAdmin();
    }

    match /college/{collegeCode} {
      allow read: if isSignedIn();
      allow create, update, delete: if isSuperAdmin();
    }

    match /helpTickets/{ticketId} {
      allow read: if isSuperAdmin() ||
        (isCollegeAdmin() && ticketOwnedByRequester(resource.data));
      allow create: if isCollegeAdmin() &&
        ticketOwnedByRequester(request.resource.data) &&
        (request.resource.data.createdByRole is string &&
          request.resource.data.createdByRole.lower() == "collegeadmin");
      allow update: if isSuperAdmin() ||
        (isCollegeAdmin() &&
          ticketOwnedByRequester(resource.data) &&
          ticketOwnedByRequester(request.resource.data));
      allow delete: if isSuperAdmin();

      match /remarks/{remarkId} {
        allow read: if isSuperAdmin() ||
          (isCollegeAdmin() &&
            ticketOwnedByRequester(
              get(/databases/$(database)/documents/helpTickets/$(ticketId)).data
            ));
        allow create: if isSuperAdmin() ||
          (isCollegeAdmin() &&
            (request.resource.data.authorUid is string &&
              request.resource.data.authorUid == request.auth.uid) &&
            ticketOwnedByRequester(
              get(/databases/$(database)/documents/helpTickets/$(ticketId)).data
            ));
        allow update, delete: if isSuperAdmin();
      }
    }
  }
}
